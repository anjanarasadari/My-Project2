select Type,Name
    -> from Room
    -> inner join Hotel on Room.Hotel_No=Hotel.Hotel_No
    -> where Type="single" or Type="double" and
    -> Price<=6500
    -> order by Price desc;

2)
 select sum(Price),Name
    -> from Room
    -> inner join Hotel on Room.Hotel_No=Hotel.Hotel_No
    -> where Type="single" and Name="Arika Villa";

3)
select Room.Room_No,Room.Hotel_No,Type,SUM(Price),Hotel.Name,Guest.Name
    -> from (((Hotel
    -> inner join Booking on Hotel.Hotel_No=Booking.Hotel_No)
    -> inner join Room on Room.Room_No=Booking.Room_No)
    -> inner join Guest on Booking.Guest_No=Guest.Guest_No)
    -> where Hotel.Name="Anilana Nilaveli" AND Date_To='2018-05-31';
